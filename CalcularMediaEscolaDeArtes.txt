algoritmo "EscolaDeArtes"
var
      AVp, AVs, Media: Real
inicio
      Escreva ("Digite a nota da AV1: ")
      Leia (AVp)
      Escreva ("Digite a nota da AV2: ")
      Leia (AVs)
      Media <- (AVp + AVs)/2
      EscrevaL ("Sua média geral é: ", Media:4:2, ".")
      Se (Media >= 7) entao
         Escreva ("Parabéns, você foi aprovado! Boas férias!")
      Senao
           Se(Media >= 5) e (Media <7) entao
            EscrevaL ("Você está em recuperação. Estude um pouco mais.")
           Senao
            Escreva ("Aluno reprovado.")
            FimSe
      FimSe
fimalgoritmo
